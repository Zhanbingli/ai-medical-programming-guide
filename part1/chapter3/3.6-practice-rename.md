    """DICOMæ–‡ä»¶æ‰¹é‡é‡å‘½åå·¥å…·"""

    def __init__(self, source_dir, output_dir=None):
        """
        åˆå§‹åŒ–

        å‚æ•°:
            source_dir: æºæ–‡ä»¶ç›®å½•
            output_dir: è¾“å‡ºç›®å½•(Noneè¡¨ç¤ºåŸåœ°é‡å‘½å)

        self.source_dir = Path(source_dir)
        self.output_dir = Path(output_dir) if output_dir else self.source_dir
        self.log_file = self.output_dir / "rename_log.txt"
        self.errors = []

    def extract_info(self, dicom_file):
        """ä»DICOMæ–‡ä»¶æå–ä¿¡æ¯"""
        try:

                "patient_age": str(ds.get("PatientAge", "æœªçŸ¥")),
                "study_date": str(ds.get("StudyDate", "æœªçŸ¥")),
                "modality": str(ds.get("Modality", "æœªçŸ¥")),
                "study_desc": str(ds.get("StudyDescription", "")),
                "series_number": str(ds.get("SeriesNumber", "001"))
            }

            return info

        except Exception as e:
            self.errors.append(f"è¯»å–å¤±è´¥: {dicom_file.name} - {str(e)}")
            return None

    def generate_new_name(self, info, original_name):
        """ç”Ÿæˆæ–°æ–‡ä»¶å"""

        # æ¸…ç†æ‚£è€…å§“å(å»é™¤ç‰¹æ®Šå­—ç¬¦)
        name = info["patient_name"].replace("^", "_")
        name = "".join(c for c in name if c.isalnum() or c in ["_", "-"])

        # å¹´é¾„å¤„ç†
        age = info["patient_age"].replace("Y", "")  # å»é™¤Yåç¼€

        # æ—¥æœŸæ ¼å¼åŒ–
        date_str = info["study_date"]
        if len(date_str) == 8:  # YYYYMMDD
            date_str = f"{date_str[:4]}{date_str[4:6]}{date_str[6:]}"

        # æ£€æŸ¥ç±»å‹æ˜ å°„
        modality_map = {
            "CT": "CT",
            "MR": "MRI",
            "CR": "Xå…‰",
            "DX": "Xå…‰",
            "US": "è¶…å£°"
        }
        exam_type = modality_map.get(info["modality"], info["modality"])

        # ç»„åˆæ–°æ–‡ä»¶å
        parts = [name, f"{age}å²", exam_type]

        if info["study_desc"]:
            parts.append(info["study_desc"][:10])  # é™åˆ¶é•¿åº¦

        parts.append(date_str)
        parts.append(info["series_number"])

        new_name = "_".join(parts) + ".dcm"

        # å¤„ç†æ–‡ä»¶åä¸­çš„éæ³•å­—ç¬¦
        illegal_chars = ['<', '>', ':', '"', '/', '\\', '|', '?', '*']
        for char in illegal_chars:
            new_name = new_name.replace(char, '_')

        return new_name

    def rename_files(self, dry_run=False):
        """
        æ‰¹é‡é‡å‘½å

        å‚æ•°:
            dry_run: Trueè¡¨ç¤ºåªæ¨¡æ‹Ÿ,ä¸å®é™…é‡å‘½å
        """
        # æŸ¥æ‰¾æ‰€æœ‰DICOMæ–‡ä»¶
        dicom_files = list(self.source_dir.glob("*.dcm"))

        if not dicom_files:
            print(f"é”™è¯¯: åœ¨{self.source_dir}ä¸­æœªæ‰¾åˆ°.dcmæ–‡ä»¶")
            return

        print(f"æ‰¾åˆ°{len(dicom_files)}ä¸ªDICOMæ–‡ä»¶\n")
        print("="*80)

        # åˆ›å»ºè¾“å‡ºç›®å½•

            self.output_dir.mkdir(parents=True, exist_ok=True)

        success_count = 0
        skip_count = 0

        log_lines = []
        log_lines.append(f"DICOMé‡å‘½åæ—¥å¿— - {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        log_lines.append("="*80)

        for dicom_file in dicom_files:
            # æå–ä¿¡æ¯
            info = self.extract_info(dicom_file)

            if info is None:
                skip_count += 1
                continue

            # ç”Ÿæˆæ–°æ–‡ä»¶å
            new_name = self.generate_new_name(info, dicom_file.name)
            new_path = self.output_dir / new_name

            # å¤„ç†æ–‡ä»¶åå†²çª
            if new_path.exists():
                base = new_path.stem
                ext = new_path.suffix
                counter = 1
                while new_path.exists():
                    new_name = f"{base}_{counter:03d}{ext}"
                    new_path = self.output_dir / new_name
                    counter += 1

            # é‡å‘½å
            log_line = f"{dicom_file.name} â†’ {new_name}"

            if dry_run:
                print(f"[æ¨¡æ‹Ÿ] {log_line}")
            else:
                try:
                    if self.output_dir == self.source_dir:
                        # åŸåœ°é‡å‘½å
                        dicom_file.rename(new_path)
                    else:
                        # å¤åˆ¶åˆ°æ–°ç›®å½•
                        import shutil
                        shutil.copy2(dicom_file, new_path)

                    print(f"âœ… {log_line}")
                    success_count += 1
                except Exception as e:
                    error_msg = f"âŒ å¤±è´¥: {dicom_file.name} - {str(e)}"
                    print(error_msg)
                    self.errors.append(error_msg)

            log_lines.append(log_line)

        # ä¿å­˜æ—¥å¿—
        if not dry_run:
            self.log_file.write_text("\n".join(log_lines), encoding="utf-8")

        # æ±‡æ€»
        print("="*80)
        print(f"\né‡å‘½å{'æ¨¡æ‹Ÿ' if dry_run else 'å®Œæˆ'}!")
        print(f"æˆåŠŸ: {success_count}ä¸ª")
        print(f"è·³è¿‡: {skip_count}ä¸ª")
        print(f"é”™è¯¯: {len(self.errors)}ä¸ª")

        if self.errors:
            print("\né”™è¯¯è¯¦æƒ…:")
            for error in self.errors:
                print(f"  {error}")

        if not dry_run:
            print(f"\næ—¥å¿—å·²ä¿å­˜: {self.log_file}")

# ä½¿ç”¨ç¤ºä¾‹
if __name__ == "__main__":
    # å…ˆæ¨¡æ‹Ÿè¿è¡Œ,æ£€æŸ¥æ•ˆæœ
    renamer = DICOMRenamer("DICOM_files", "renamed_files")
